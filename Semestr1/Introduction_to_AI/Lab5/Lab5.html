def getConcordanceGain(gA, gB, q, p):
    if (gB<=gA+q):
        return 1
    elif (gB<=gA+p):
        return (p-(gB-gA))/(p-q)
    else:
        return 0

def getComprehensiveConcordance(A, B, criteria, parameters):
    concordance = 0.0
    sum_weights=0.0
    for i in range (len(criteria)):
        if (parameters[criteria[i]]['type']=='gain'):
            concordance+=parameters[criteria[i]]['weight']*getConcordanceGain(A[criteria[i]], B[criteria[i]], parameters[criteria[i]]['q'], parameters[criteria[i]]['p'])
        else:
            concordance+=parameters[criteria[i]]['weight']*getConcordanceCost(A[criteria[i]], B[criteria[i]], parameters[criteria[i]]['q'], parameters[criteria[i]]['p'])
        sum_weights+=parameters[criteria[i]]['weight']
        
    return concordance / sum_weights

def getConcordanceMatrix(data, criteria, parameters, majority_treshold=0.7):
    concordance_matrix = np.zeros((len(data),len(data)))
    for A_idx, A_row in data.iterrows():
        for B_idx, B_row in data.iterrows():
            if (A_idx!=B_idx):
                conc=getComprehensiveConcordance(data.loc[A_idx], B_row, criteria, parameters)
                if conc>=majority_treshold: concordance_matrix[A_idx][B_idx]=1.
                else: concordance_matrix[A_idx][B_idx]=0.
            continue
              
    return concordance_matrix

def getOutrankingMatrix(data, criteria, parameters, majority_treshold):
    concordance_matrix = getConcordanceMatrix(data, criteria, parameters, majority_treshold)
    discordance_matrix = getDiscordanceMatrix(data, criteria, parameters)
    n = len(data)
    outranking_matrix = np.zeros((n,n))
    for i in range (n):
        for j in range (n):
            if(concordance_matrix[i][j]==1 and discordance_matrix[i][j]==0):
                outranking_matrix[i][j]=1
    return outranking_matrix

def toAdjacencyList(outranking_matrix):
    n = len(outranking_matrix)
    graph = {i:[] for i in range(n)}
    for i in range (n):
        for j in range (n):
            if (outranking_matrix[i][j]==1):
                graph[i].append(j)
    return graph

def getReverseGraph(graph):
    n = len(graph)
    reverse_graph = {i:[] for i in graph}
    for i in range (n):
        for j in range (len(graph[i])):
            reverse_graph[graph[i][j]].append(i)
    return reverse_graph

3.2) Now, complete the below function for finding a graph kernel. This algorithm should proceed in the following way:

  1. Find non-outranked vertices. Add them to the kernel.
  2. Remove the already found vertices from the graph and these vertices which are surpassed by them.
  3. Repeat (go to 1) until all vertices are removed from the graph.

You can use the auxiliary reverse_graph structure. It can be helpfull for finding non-outranked vertices.

def Kernel(graph):
    reverse_graph = getReverseGraph(graph)
    kernel = []
    not_in_kernel=[]
    while (len(kernel)+len(not_in_kernel)<len(graph)):
        for i in list(reverse_graph.keys()):
            if (reverse_graph[i]==[] and i not in kernel and i not in not_in_kernel): #step 1
                kernel.append(i)
                for j in list(reverse_graph.keys()): #step 2a
                    if (i in reverse_graph[j]): 
                        reverse_graph[j].pop(reverse_graph[j].index(i))
                        if (j not in not_in_kernel):
                            not_in_kernel.append(j)
                            for m in list(reverse_graph.keys()): #step 2b
                                if (j in reverse_graph[m]):
                                    reverse_graph[m].pop(reverse_graph[m].index(j))
    return kernel
